#Day-20-Remove K Digits

class Solution {
public:
    string removeKdigits(string num, int k) {
        string st = "";
        
        for (char digit : num) {
            while (!st.empty() && k > 0 && st.back() > digit) {
                st.pop_back();
                k--;
            }

            st.push_back(digit);
        }
        
        while (k > 0) {
            st.pop_back();
            k--;
        }
        
        if (st.empty()) return "0";
        
        int i = 0;
        while (i < st.length() - 1 && st[i] == '0') i++;
        return st.substr(i);
    }
};
