class Solution {
public:
    int diameterOfBinaryTree(TreeNode* root) {
    int ans=0;
    function<int(TreeNode*)> h=[&](TreeNode* node)->int{
        if(!node)return 0;
        int l=h(node->left),r=h(node->right);
        ans=max(ans,l+r); return 1+max(l,r);
    };
    h(root); return ans;
}
